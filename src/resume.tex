\ExplSyntaxOn
    % get environment variable.
    % example: \getenv{HOME}
    % reference: https://tex.stackexchange.com/questions/62010/can-i-access-system-environment-variables-from-latex-for-instance-home
    \NewDocumentCommand{\getenv}{om}{
        \sys_get_shell:nnN { kpsewhich ~ --var-value ~ #2 }
        { \cctab_select:N \c_str_cctab } \l_tmpa_tl
        \tl_trim_spaces:N \l_tmpa_tl
        \IfNoValueTF { #1 }
        { \tl_use:N \l_tmpa_tl }
        { \tl_set_eq:NN #1 \l_tmpa_tl }
    }

    % check if env variable is set and non-blank.
    % example: \ifenvsetTF{PHONE_NUMBER}
    % reference: https://tex.stackexchange.com/questions/548836/switch-based-on-whether-or-not-an-environment-variable-is-set
    \tl_const:Nn \c_getenv_par_tl { \par }
    \NewDocumentCommand{\ifenvsetTF}{mmm}{
        \sys_get_shell:nnN { kpsewhich ~ --var-value ~ #1 } { } \l_tmpa_tl
        \tl_if_eq:NNTF \l_tmpa_tl \c_getenv_par_tl { #3 } { #2 }
    }
\ExplSyntaxOff
